1. Web — Next.js 16 / React 19

The web client uses the official @heroiclabs/nakama-js SDK plus a thin wrapper at web/lib/nakama/web-client.ts. The board component (TournamentsBoard.tsx) and detail card already exist; you mostly wire them into a route.

1.1 Install + env

npm i @heroiclabs/nakama-js

# .env.local
NEXT_PUBLIC_NAKAMA_HOST=nakama.intelli-verse-x.ai
NEXT_PUBLIC_NAKAMA_PORT=443
NEXT_PUBLIC_NAKAMA_USE_SSL=true
NEXT_PUBLIC_NAKAMA_KEY=defaultkey   # client key only, NOT the server http_key

1.2 Mint a session (anonymous-first, merge on sign-in)

// web/lib/nakama/web-client.ts (already exists in repo)
import { Client } from "@heroiclabs/nakama-js";

const client = new Client(
  process.env.NEXT_PUBLIC_NAKAMA_KEY!,
  process.env.NEXT_PUBLIC_NAKAMA_HOST!,
  process.env.NEXT_PUBLIC_NAKAMA_PORT!,
  process.env.NEXT_PUBLIC_NAKAMA_USE_SSL === "true",
);

let session;
export async function ensureSession() {
  if (session) return session;
  // ghost user = device-bound anonymous; merges on Google sign-in via authenticateGoogle()
  const deviceId = crypto.randomUUID();
  session = await client.authenticateDevice(deviceId, true);
  return session;
}

export async function rpc<T = any>(name: string, payload: object) {
  const s = await ensureSession();
  const r = await client.rpc(s, name, payload);
  return { ok: true, data: r.payload as T };
}

// app/tournaments/page.tsx

1.4 Wire the Enter button

// inside TournamentDetail.tsx — already implemented
const onEnter = async () => {
  const idempotency_key = crypto.randomUUID();
  const r = await rpc("tournament_enter", { slug, idempotency_key });
  if (r.data?.ok) {
    setEntryId(r.data.entry_id);
    setPotBc(r.data.pot_bc);
    showToast("You're in. Pot is now " + r.data.pot_bc + " BC");
  } else if (r.data?.error === "insufficient_balance") {
    openBuyBcSheet();
  } else if (r.data?.error === "entry_blocked_in_us_state") {
    showAmoeUpsell();
  }
};

2. Unity — C# (Built-in pipeline)

Unity uses Nakama.Client (the official C# SDK, already in Assets/Plugins/Nakama/). The QuizVerse SDK wrapper at Assets/_QuizVerse/Scripts/Backend/NakamaService.cs exposes a typed RpcAsync<T>() helper.

2.1 List tournaments

using Nakama;
using Nakama.TinyJson;

public async UniTask<TournamentListResponse> ListTournamentsAsync()
{
    var rpcResult = await NakamaService.Instance.RpcAsync(
        "tournament_list",
        new { }
    );
    return rpcResult.Payload.FromJson<TournamentListResponse>();
}

[System.Serializable]
public class TournamentListResponse {
    public bool success;
    public TournamentListData data;
}
[System.Serializable]
public class TournamentListData {
    public TournamentSummary[] tournaments;
}
[System.Serializable]
public class TournamentSummary {
    public string slug;
    public string name;
    public string format;
    public string status;
    public int pot_bc;
    public int entries_count;
    public int entry_fee_bc;
}

2.2 Enter a tournament

public async UniTask<EnterResult> EnterAsync(string slug)
{
    var key = System.Guid.NewGuid().ToString();
    var rpc = await NakamaService.Instance.RpcAsync(
        "tournament_enter",
        new { slug = slug, idempotency_key = key }
    );
    return rpc.Payload.FromJson<EnterResult>();
}

2.3 Submit a pack result

public async UniTask SubmitPackAsync(string slug, string packId, int[] answers, int latencyMs)
{
    await NakamaService.Instance.RpcAsync(
        "tournament_submit_pack_result",
        new { slug, pack_id = packId, answers, client_latency_ms = latencyMs }

2.4 Live pot via socket

public async UniTask SubscribeToTournamentAsync(string slug, System.Action<int> onPot)
{
    var socket = NakamaService.Instance.Socket;
    var channel = await socket.JoinChatAsync("tournament:" + slug, ChannelType.Room);
    socket.ReceivedChannelMessage += (msg) => {
        var data = JsonUtility.FromJson<PotMsg>(msg.Content);
        if (data.op == 1001) onPot(data.pot_bc);
    };
}
[System.Serializable] class PotMsg { public int op; public int pot_bc; }

3. Backend — direct Nakama JS module

Most server-side code is already shipped in data/modules/src/tournaments/. You only need backend work if you're adding a new tournament format, a new RPC, or wiring a new microservice.

3.1 Adding a new tournament to the launch slate

Edit data/modules/src/p2e/tournament_economy.ts:

// Append to LAUNCH_SLATE
{
  slug: "your-new-tournament",
  name: "Your New Tournament",
  description: "What it is.",
  topic_tag: "your_topic",
  format: "classic",
  format_ui_variant: "classic-pot",
  pre_enroll_start_iso: PRE_ENROLL_OPEN_ISO,
  open_start_iso: WAVE_1_OPEN_ISO,
  end_iso: WAVE_1_WEEK_END_ISO,
  entry_fee_bc: 250,
  rake_pct: HOUSE_RAKE_PCT,
  pot_seed_bc: 15000,
  pot_split_top_n: CLASSIC_POT_SPLIT_TOP_N,
  countries_allowed: "ALL",
  min_age: MIN_AGE,
  amoe: AMOE_CLASSIC,
  badge_emoji: "🆕",
}

Then rebuild and deploy:

cd data/modules
npm run build      # tsc + postbuild → data/modules/index.js
git add . && git commit -m "feat(tournaments): seed your-new-tournament"
git push origin master

The CI pipeline (intelliverse-nakama) will build, push to ECR, and roll out to all 8 EKS pods. The cron tournament_cron_tick will seed the new slug on the next tick.

3.2 Running locally

cd data/modules
npm install
npm run build       # outputs to data/modules/index.js
docker compose up   # boots Nakama with the JS module mounted

3.3 Adding a new RPC

Three steps:

  1. Implement in data/modules/src/tournaments/rpcs.ts as a top-level function: function rpcMyNew(ctx, logger, nk, payload): { ok: true, ... }.
  2. Add initializer.registerRpc("my_new_rpc", rpcMyNew) at the bottom of the same file.
  3. Run npm run build and verify the count went up: grep -c registerRpc data/modules/build/index.js.

Goja gotcha: the RPC handler MUST be a top-level function declaration, not a function expression or method. The AST walker rejects nested or assigned handlers. See postbuild.js for the auto-injection that satisfies this constraint for tournament RPCs.
